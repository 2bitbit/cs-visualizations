Windows x64 ABI 核心机制：局部变量与对齐

场景: 含有 2 个局部变量，并调用 func(1, 2, 3, 4, 5, 6)

反汇编指令流 (MSVC 语法) RIP 地址

test: // 调用者 Caller
; 入口 RSP = 0x15FF08 (因 call 指令压入了返回地址，目前非 16 字节对齐)
sub rsp, 48h ; 一次性打包分配 72 字节空间 0x...1030
; 1. 初始化局部变量
mov qword ptr [rsp+30h], 99 ; 局部变量 A 0x...1034
mov qword ptr [rsp+38h], 88 ; 局部变量 B 0x...103D
; 2. 准备子函数栈参数
mov qword ptr [rsp+28h], 6 ; 第 6 个参数 0x...1046
mov qword ptr [rsp+20h], 5 ; 第 5 个参数 0x...104E
; 3. 准备寄存器参数
mov r9d, 4 ; 第 4 个参数 (R9) 0x...1056
mov r8d, 3 ; 第 3 个参数 (R8) 0x...105C
mov edx, 2 ; 第 2 个参数 (RDX) 0x...1062
mov ecx, 1 ; 第 1 个参数 (RCX) 0x...1067
call func ; RSP 目前为 0x...C0 (满足 16 字节对齐!) 0x...106A
add rsp, 48h ; 销毁栈帧 (恢复 RSP) 0x...106F
ret ; 返回至上一级调用者 0x...1073
func: // 被调用者 (-O0)
mov [rsp+8], rcx ; Spill RCX 到影子空间 0x...2000
mov [rsp+10h], rdx ; Spill RDX 到影子空间 0x...2005
mov [rsp+18h], r8 ; Spill R8 到影子空间 0x...200A
mov [rsp+20h], r9 ; Spill R9 到影子空间 0x...200F
... ; 函数核心逻辑 ...
ret ; 弹出返回地址 0x...2050

72字节计算公式:
  • 局部变量: 16 字节
  • 对齐填充: 8 字节
  • 超出参数: 16 字节
  • 影子空间: 32 字节 (强制)
总计 = 72 字节 (0x48)
