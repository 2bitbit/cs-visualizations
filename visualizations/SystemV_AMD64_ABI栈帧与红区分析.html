反汇编指令流 (AT&T 语法) RIP 地址

test: // 调用者 Caller
# 入口 RSP = 0x...E008 (非 16 字节对齐)
pushq %rbp # 保存旧基址 (RSP 变 0x...E000，对齐) 0x...1020
movq %rsp, %rbp # 确立帧基址 0x...1021
# 1. 动态分配局部变量
subq $16, %rsp # 腾出 16 字节 (RSP 依然 16 字节对齐!) 0x...1024
movq $99, -8(%rbp) # 局部变量 B 0x...1028
movq $88, -16(%rbp) # 局部变量 A 0x...1030
# 2. 动态压栈参数 (从右到左)
pushq $8 # 第 8 个参数 (RSP 破除对齐) 0x...1038
pushq $7 # 第 7 个参数 (RSP 再次 16 字节对齐!) 0x...103A
# 3. 填充寄存器参数
movq ... %r9, %r8 ... %rdi # 准备 6 个寄存器参数 0x...103C
call func # 完美满足对齐要求，跳转执行 0x...1050
# 4. 调用者清理现场
addq $16, %rsp # 调用者主动弹出 2 个栈参数 0x...1055
leave # 销毁 test 栈帧 0x...1059
ret # 返回 0x...105A
func: // 被调用者 (叶子函数)
pushq %rbp 0x...2000
movq %rsp, %rbp 0x...2001
# 直接使用 Red Zone，无 subq %rsp 分配！
movq %rdi, -8(%rbp) # Spill RDI 到红区 0x...2004
movq %rsi, -16(%rbp) # Spill RSI 到红区 0x...2008
... # 函数核心逻辑 ...
# 5. 被调者返回
popq %rbp # 恢复调用者的 RBP 0x...2050
ret # 弹栈返回 test 0x...2051

对齐与红区解析：
  • 无 Shadow Space：Linux 不要求预留 32 字节空地。
  • 动态对齐法：局部变量(16B) + 栈参数(16B) = 32B。这种搭配使得 call 之前恰好维持 16 字节对齐。
  • 红区 (Red Zone)：叶子函数可以直接利用 RSP 下方的 128 字节，避免了不必要的 sub %rsp 指令。
